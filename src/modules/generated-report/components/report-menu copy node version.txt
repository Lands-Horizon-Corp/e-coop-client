const renderTrigger = (item: ReportItem) => {
    const Icon = item.icon
    return (
        <Button
            className="group h-8 w-full justify-start gap-2 rounded border-border/60 bg-card px-2 text-left hover:border-primary/50 hover:bg-accent"
            size="sm"
            variant="outline"
        >
            <Icon className="size-3.5 shrink-0 text-primary" />
            <span className="truncate text-xs font-medium">{item.label}</span>
        </Button>
    )
}

const ReportTile = ({ item }: { item: ReportItem }) => {
    const Component = item.component
    const trigger = renderTrigger(item)
    if (!Component) return <div>{trigger}</div>
    return <Component trigger={trigger} />
}

const countLeaves = (node: ReportNode): number =>
    node.type === 'report'
        ? 1
        : node.children.reduce((s, c) => s + countLeaves(c), 0)

const NestedGroup = ({ group, depth }: { group: ReportGroup; depth: number }) => {
    const Icon = group.icon
    const leaves = group.children.filter((c): c is ReportItem => c.type === 'report')
    const subs = group.children.filter((c): c is ReportGroup => c.type === 'group')
    const total = countLeaves(group)

    return (
        <div
            className="rounded-md border border-border/60 bg-muted/30 p-2"
            style={{ marginLeft: depth > 1 ? 4 : 0 }}
        >
            <div className="mb-1.5 flex items-center gap-1.5">
                <Icon className="size-3 text-muted-foreground" />
                <h3 className="text-[10px] font-semibold uppercase tracking-wide text-muted-foreground">
                    {group.label}
                </h3>
                <div className="ml-auto h-px flex-1 bg-border" />
                <span className="font-mono text-[10px] text-muted-foreground">{total}</span>
            </div>

            {leaves.length > 0 && (
                <div className="grid grid-cols-1 gap-1 sm:grid-cols-2">
                    {leaves.map((r) => (
                        <ReportTile item={r} key={r.persistKey ?? r.label} />
                    ))}
                </div>
            )}

            {subs.length > 0 && (
                <div className={`grid ${leaves.length > 0 ? 'mt-2' : ''}`}>
                    {subs.map((s) => (
                        <NestedGroup depth={depth + 1} group={s} key={s.label} />
                    ))}
                </div>
            )}
        </div>
    )
}

type FlatLeaf = {
    item: ReportItem
    path: string[]
    pathIcons: React.ElementType[]
}

const flatten = (
    nodes: ReportNode[],
    path: string[] = [],
    pathIcons: React.ElementType[] = []
): FlatLeaf[] =>
    nodes.flatMap((n) =>
        n.type === 'report'
            ? [{ item: n, path, pathIcons }]
            : flatten(n.children, [...path, n.label], [...pathIcons, n.icon])
    )

const rebuildTree = (leaves: FlatLeaf[]): ReportGroup[] => {
    const roots: ReportGroup[] = []
    for (const leaf of leaves) {
        let currentChildren: ReportNode[] = roots as unknown as ReportNode[]
        for (let i = 0; i < leaf.path.length; i++) {
            const label = leaf.path[i]
            let node = currentChildren.find(
                (c): c is ReportGroup => c.type === 'group' && c.label === label
            )
            if (!node) {
                node = {
                    type: 'group',
                    label,
                    icon: leaf.pathIcons[i],
                    children: [],
                }
                currentChildren.push(node)
            }
            currentChildren = node.children
        }
        currentChildren.push(leaf.item)
    }
    return roots
}

function ReportsMenu() {
    const [search, setSearch] = useState('')
    const inputRef = useRef<HTMLInputElement>(null)

    useEffect(() => {
        inputRef.current?.focus()
    }, [])

    const flatLeaves = useMemo(() => flatten(tree), [])

    const fuse = useMemo(
        () =>
            new Fuse(flatLeaves, {
                keys: ['item.label', 'path'],
                threshold: 0.3,
            }),
        [flatLeaves]
    )

    const filteredTree = useMemo<ReportGroup[]>(() => {
        if (!search.trim()) return tree
        const matched = fuse.search(search).map((r) => r.item)
        return rebuildTree(matched)
    }, [search, fuse])

    const total = flatLeaves.length

    return (
        <div className="min-h-screen w-fit  bg-background">
            <header className="sticky top-0 z-10 border-b">
                <div className="container flex flex-col gap-3 py-4 md:flex-row md:items-center md:justify-between">
                    <div className="flex items-center gap-3">
                        <span className="flex size-9 items-center justify-center rounded-md bg-primary text-primary-foreground">
                            <LayoutGrid className="size-4" />
                        </span>
                        <div>
                            <h1 className="text-lg font-bold leading-tight">Reports Menu</h1>
                            <p className="text-xs text-muted-foreground">{total} reports available</p>
                        </div>
                    </div>
                    <div className="relative w-full md:w-72">
                        <Search className="pointer-events-none absolute left-3 top-1/2 size-4 -translate-y-1/2 text-muted-foreground" />
                        <Input
                            className="h-9 pl-9"
                            onChange={(e) => setSearch(e.target.value)}
                            placeholder="Search reports..."
                            ref={inputRef}
                            value={search}
                        />
                    </div>
                </div>
            </header>

            <main className="container py-4">
                {filteredTree.length === 0 && (
                    <Card>
                        <CardContent className="py-12 text-center text-sm text-muted-foreground">
                            No reports match "{search}"
                        </CardContent>
                    </Card>
                )}
                <div className="grid gap-4 lg:grid-cols-2">
                    {filteredTree.map((group) => {
                        const GroupIcon = group.icon
                        const groupTotal = countLeaves(group)
                        const leaves = group.children.filter(
                            (c): c is ReportItem => c.type === 'report'
                        )
                        const subs = group.children.filter(
                            (c): c is ReportGroup => c.type === 'group'
                        )
                        return (
                            <Card className="overflow-hidden" key={group.label}>
                                <CardHeader className="flex-row items-center gap-3 space-y-0 border-b bg-muted/40 py-2.5">
                                    <span className="flex size-7 items-center justify-center rounded-md bg-primary text-primary-foreground">
                                        <GroupIcon className="size-3.5" />
                                    </span>
                                    <CardTitle className="text-sm">{group.label}</CardTitle>
                                    <Badge className="ml-auto font-mono text-[10px]" variant="secondary">
                                        {groupTotal}
                                    </Badge>
                                </CardHeader>
                                <CardContent className="space-y-4 p-2 gap-4 grid h-fit grid-cols-2">
                                    {leaves.length > 0 && (
                                        <div className="grid grid-cols-2 h-fit gap-1">
                                            {leaves.map((r) => (
                                                <ReportTile item={r} key={r.persistKey ?? r.label} />
                                            ))}
                                        </div>
                                    )}
                                    {subs.map((sub) => (
                                        <NestedGroup depth={1} group={sub} key={sub.label} />
                                    ))}
                                </CardContent>
                            </Card>
                        )
                    })}
                </div>
            </main>
        </div>
    )
}

export default ReportsMenu
